64       public Packet250CustomPayload getMorphInfoAsPacket() 
65       { 
66           ByteArrayOutputStream bytes = new ByteArrayOutputStream(); 
67           DataOutputStream stream = new DataOutputStream(bytes); 
68           try 
69           { 
70               stream.writeByte(0); //id 
71               stream.writeUTF(playerName); 
72                
73               stream.writeBoolean(morphing); 
74               stream.writeInt(morphProgress); 
75    
76               stream.writeBoolean(prevState != null); 
77               if(prevState != null) 
78               { 
79                   Morph.writeNBTTagCompound(prevState.getTag(), stream); 
80               } 
81               stream.writeBoolean(nextState != null); 
82               if(nextState != null) 
83               { 
84                   Morph.writeNBTTagCompound(nextState.getTag(), stream); 
85               } 
86                
87               stream.writeBoolean(flying); 
88           } 
89           catch(IOException e) 
90           { 
91                
92           } 
93           return new Packet250CustomPayload("Morph", bytes.toByteArray()); 
94       }

96       public void writeNBT(NBTTagCompound tag) 
97       { 
98           tag.setString("playerName", playerName); 
99            
100          tag.setInteger("dimension", nextState.entInstance.dimension); 
101           
102          tag.setCompoundTag("nextState", nextState.getTag()); 
103           
104          tag.setBoolean("isFlying", flying); 
105      } 
106       
107      public void readNBT(NBTTagCompound tag) 
108      { 
109          playerName = tag.getString("playerName"); 
110           
111          World dimension = DimensionManager.getWorld(tag.getInteger("dimension")); 
112           
113          if(dimension == null) 
114          { 
115              dimension = DimensionManager.getWorld(0); 
116          } 
117           
118          nextState = new MorphState(dimension, playerName, playerName, null, false); 
119           
120          nextState.readTag(dimension, tag.getCompoundTag("nextState")); 
121           
122          morphing = true; 
123          morphProgress = 80; 
124           
125          ArrayList<Ability> newAbilities = AbilityHandler.getEntityAbilities(nextState.entInstance.getClass()); 
126          morphAbilities = new ArrayList<Ability>(); 
127          for(Ability ability : newAbilities) 
128          { 
129              try 
130              { 
131                  morphAbilities.add(ability.clone()); 
132              } 
133              catch(Exception e1) 
134              { 
135              } 
136          } 
137           
138          flying = tag.getBoolean("isFlying"); 
139      }
